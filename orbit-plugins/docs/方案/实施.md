```go
package sdk

type FetchRequest struct {
    Route  string            `json:"route"`
    Params map[string]string `json:"params"`

type FeedResult struct {
    Title       string     `json:"title"`
    Description string     `json:"description,omitempty"`
    Items       []FeedItem `json:"items"`
}

type FeedItem struct {
    ID          string   `json:"id"`
    Title       string   `json:"title"`
    URL         string   `json:"url"`
    Content     string   `json:"content,omitempty"`
    Summary     string   `json:"summary,omitempty"`
    Author      string   `json:"author,omitempty"`
    Cover       string   `json:"cover,omitempty"`
    PublishedAt string   `json:"published_at"`
    Tags        []string `json:"tags,omitempty"`
}

type Manifest struct {
    ID          string        `json:"id"`
    Name        string        `json:"name"`
    Version     string        `json:"version"`
    Description string        `json:"description"`
    Icon        string        `json:"icon,omitempty"`
    Routes      []RouteConfig `json:"routes"`
}

type RouteConfig struct {
    Path        string     `json:"path"`
    Name        string     `json:"name"`
    Description string     `json:"description"`
    Params      []ParamDef `json:"params,omitempty"`
}

type ParamDef struct {
    Name     string `json:"name"`
    Desc     string `json:"desc"`
    Required bool   `json:"required"`
    Default  string `json:"default,omitempty"`
}
```

```go
// sdk/sdk.go
package sdk

import (
    "encoding/json"
    "os"
)

// Plugin 是每个插件必须实现的接口
type Plugin interface {
    Manifest() *Manifest
    Fetch(req *FetchRequest) (*FeedResult, error)
}

type envelope struct {
    Action string          `json:"action"`
    Data   json.RawMessage `json:"data"`
}

type response struct {
    OK    bool   `json:"ok"`
    Data  any    `json:"data,omitempty"`
    Error string `json:"error,omitempty"`
}

// Run 是插件 main() 里唯一需要调用的函数
func Run(p Plugin) {
    var env envelope
    if err := json.NewDecoder(os.Stdin).Decode(&env); err != nil {
        writeError(err)
        return
    }

    switch env.Action {
    case "manifest":
        writeOK(p.Manifest())

    case "fetch":
        var req FetchRequest
        if err := json.Unmarshal(env.Data, &req); err != nil {
            writeError(err)
            return
        }
        result, err := p.Fetch(&req)
        if err != nil {
            writeError(err)
            return
        }
        writeOK(result)

    default:
        writeError(fmt.Errorf("unknown action: %s", env.Action))
    }
}

func writeOK(data any) {
    json.NewEncoder(os.Stdout).Encode(response{OK: true, Data: data})
}

func writeError(err error) {
    json.NewEncoder(os.Stdout).Encode(response{OK: false, Error: err.Error()})
}
```

## 掘金插件（完整独立 module）

```go
// plugins/juejin/go.mod
module github.com/orbit-app/plugin-juejin

go 1.21

require github.com/orbit-app/plugin-sdk v0.1.0

// 本地开发时替换为本地路径
replace github.com/orbit-app/plugin-sdk => ../../sdk
```

```go
// plugins/juejin/main.go
package main

import (
    "encoding/json"
    "fmt"
    "io"
    "net/http"
    "strconv"
    "time"

    sdk "github.com/orbit-app/plugin-sdk"
)

func main() {
    sdk.Run(&JuejinPlugin{})
}

// ---- 插件实现 ----

type JuejinPlugin struct{}

func (p *JuejinPlugin) Manifest() *sdk.Manifest {
    return &sdk.Manifest{
        ID:          "juejin",
        Name:        "掘金",
        Version:     "1.0.0",
        Description: "掘金开发者社区",
        Routes: []sdk.RouteConfig{
            {
                Path:        "/juejin/trending",
                Name:        "热榜",
                Description: "掘金全站热榜",
            },
            {
                Path:        "/juejin/category/:category",
                Name:        "分类热榜",
                Description: "按分类获取热门文章",
                Params: []sdk.ParamDef{
                    {
                        Name:     "category",
                        Desc:     "frontend / backend / android / ios / ai",
                        Required: true,
                        Default:  "frontend",
                    },
                },
            },
            {
                Path:        "/juejin/tag/:tag",
                Name:        "标签文章",
                Description: "按标签获取最新文章",
                Params: []sdk.ParamDef{
                    {
                        Name:     "tag",
                        Desc:     "标签名，如 Vue.js / React / Go",
                        Required: true,
                    },
                },
            },
        },
    }
}

func (p *JuejinPlugin) Fetch(req *sdk.FetchRequest) (*sdk.FeedResult, error) {
    switch {
    case req.Route == "/juejin/trending":
        return fetchTrending()

    case req.Route == "/juejin/category/:category":
        category := req.Params["category"]
        if category == "" {
            category = "frontend"
        }
        return fetchByCategory(category)

    case req.Route == "/juejin/tag/:tag":
        tag := req.Params["tag"]
        if tag == "" {
            return nil, fmt.Errorf("tag param is required")
        }
        return fetchByTag(tag)

    default:
        return nil, fmt.Errorf("unknown route: %s", req.Route)
    }
}

// ---- 具体抓取逻辑 ----

var categoryIDMap = map[string]string{
    "frontend": "6809637767543259144",
    "backend":  "6809637769959178254",
    "android":  "6809635626879549448",
    "ios":      "6809635626661445640",
    "ai":       "6809637773935869959",
}

func fetchTrending() (*sdk.FeedResult, error) {
    url := "https://api.juejin.cn/content_api/v1/content/article_rank?category_id=1&type=hot"
    return doFetch(url, "掘金热榜", "掘金全站热门文章")
}

func fetchByCategory(category string) (*sdk.FeedResult, error) {
    cid, ok := categoryIDMap[category]
    if !ok {
        return nil, fmt.Errorf("unknown category: %s", category)
    }
    url := fmt.Sprintf(
        "https://api.juejin.cn/recommend_api/v1/article/recommend_cate_feed?cate_id=%s&cursor=0&limit=20&sort_type=3",
        cid,
    )
    return doFetch(url, fmt.Sprintf("掘金 · %s", category), "")
}

func fetchByTag(tag string) (*sdk.FeedResult, error) {
    // 先查 tag_id
    tagID, err := queryTagID(tag)
    if err != nil {
        return nil, err
    }
    url := fmt.Sprintf(
        "https://api.juejin.cn/recommend_api/v1/article/recommend_tag_feed?tag_id=%s&cursor=0&limit=20&sort_type=200",
        tagID,
    )
    return doFetch(url, fmt.Sprintf("掘金 · #%s", tag), "")
}

func queryTagID(tagName string) (string, error) {
    url := fmt.Sprintf(
        "https://api.juejin.cn/tag_api/v1/query_tag_detail?tag=%s",
        tagName,
    )
    resp, err := http.Get(url)
    if err != nil {
        return "", err
    }
    defer resp.Body.Close()
    
    body, _ := io.ReadAll(resp.Body)
    var result struct {
        Data struct {
            TagID string `json:"tag_id"`
        } `json:"data"`
    }
    json.Unmarshal(body, &result)
    
    if result.Data.TagID == "" {
        return "", fmt.Errorf("tag not found: %s", tagName)
    }
    return result.Data.TagID, nil
}

// doFetch 是通用的文章列表抓取
func doFetch(url, title, desc string) (*sdk.FeedResult, error) {
    resp, err := http.Get(url)
    if err != nil {
        return nil, fmt.Errorf("http get failed: %w", err)
    }
    defer resp.Body.Close()

    body, err := io.ReadAll(resp.Body)
    if err != nil {
        return nil, err
    }

    var raw struct {
        Data []struct {
            ArticleID   string `json:"article_id"`
            ArticleInfo struct {
                Title        string `json:"title"`
                BriefContent string `json:"brief_content"`
                CoverImage   string `json:"cover_image"`
                Ctime        string `json:"ctime"`
            } `json:"article_info"`
            AuthorUserInfo struct {
                Username string `json:"user_name"`
            } `json:"author_user_info"`
            Tags []struct {
                TagName string `json:"tag_name"`
            } `json:"tags"`
        } `json:"data"`
    }

    if err := json.Unmarshal(body, &raw); err != nil {
        return nil, fmt.Errorf("parse response failed: %w", err)
    }

    items := make([]sdk.FeedItem, 0, len(raw.Data))
    for _, d := range raw.Data {
        tags := make([]string, 0, len(d.Tags))
        for _, t := range d.Tags {
            tags = append(tags, t.TagName)
        }

        ts, _ := strconv.ParseInt(d.ArticleInfo.Ctime, 10, 64)
        publishedAt := time.Unix(ts, 0).Format(time.RFC3339)

        items = append(items, sdk.FeedItem{
            ID:          d.ArticleID,
            Title:       d.ArticleInfo.Title,
            URL:         "https://juejin.cn/post/" + d.ArticleID,
            Summary:     d.ArticleInfo.BriefContent,
            Cover:       d.ArticleInfo.CoverImage,
            Author:      d.AuthorUserInfo.Username,
            PublishedAt: publishedAt,
            Tags:        tags,
        })
    }

    return &sdk.FeedResult{
        Title:       title,
        Description: desc,
        Items:       items,
    }, nil
}
```

## Makefile（单插件 & 根目录批量）

```makefile
# plugins/juejin/Makefile
PLUGIN_NAME = juejin
OUTPUT      = ../../dist/$(PLUGIN_NAME).wasm

build:
	GOOS=wasip1 GOARCH=wasm go build -o $(OUTPUT) .

# 开发时用：实时查看输出
run:
	echo '{"action":"manifest"}' | GOOS=js go run . 

clean:
	rm -f $(OUTPUT)
```

```makefile
# 根 Makefile，一键编译所有插件
PLUGINS = juejin bilibili sspai

build-all:
	@for p in $(PLUGINS); do \
		echo "→ building $$p..."; \
		$(MAKE) -C plugins/$$p build; \
	done
	@echo "✅ all plugins built"

clean-all:
	@for p in $(PLUGINS); do \
		$(MAKE) -C plugins/$$p clean; \
	done
```

## 本地调试技巧

开发时不用每次编译 WASM，直接用 JSON pipe 测试：

```bash
# 测试 manifest
echo '{"action":"manifest"}' | go run .

# 测试 fetch
echo '{"action":"fetch","data":{"route":"/juejin/category/:category","params":{"category":"frontend"}}}' \
  | go run .

# 编译并用 wasmtime 直接跑（不依赖 Go host）
GOOS=wasip1 GOARCH=wasm go build -o juejin.wasm .
echo '{"action":"manifest"}' | wasmtime juejin.wasm
```

这样每个插件开发、测试、编译完全独立，加新插件只需新建一个目录，不影响其他任何东西。要我继续写 bilibili 或少数派的插件实现吗？
